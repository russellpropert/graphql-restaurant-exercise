# Queries and Mutations

query restaurant($id: ID!) {
  restaurant(id: $id) {
    name
    description
    dishes {
      name
      price
    }
  }
}

query restaurants {
  restaurants {
    id
    name
    description
    dishes {
      name
      price
    }
  }
}
